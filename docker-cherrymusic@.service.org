# systemd unit file for cherrymusic docker service
# Andreas Linz - klingt.net@gmail.com
# https://github.com/klingtdotnet/docker-cherrymusic
#
# Note:
#   - tasks that are prefixed with an `-` are allowed to fail

[Unit]
Description=Cherrymusic docker
Requires=docker.service
After=docker.service

[Service]
User=%I
Restart=on-failure
RestartSec=10
ExecStartPre=-/usr/bin/docker rm -f docker-cherrymusic
ExecStart=/mnt/internal/a/apps/cherrymusic/run.sh
ExecStop=/usr/bin/docker stop docker-cherrymusic
ExecStop=-/usr/bin/docker rm docker-cherrymusic

[Install]
WantedBy=multi-user.target
